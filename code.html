/*calcul mensuel*/


proc sql;

create table indicateurs_mm as

select numorg, year(ddrefenr) as annee, month(ddrefenr) as mois, /*cles de la table*/

ncoua + ncouas as nbcour_mm, /*nb de courriers*/

npieat as nbpieces_mm, /*nb pieces*/

nbvistot as nbvis_mm, /*nb visite*/

telabo+telabos as nbappabo_mm, /*nb appels aboutis*/

(teltragt+teltrsvt)/telabot as txapptra_mm, /*taux appels traites*/

(teltrag+teltrags)/(telabag+telabags) as txapptragt_mm, /*taux appels traites agent*/ 

npipso9t/(npipso9t+npipso1t) as minima_mm, /*delai de traitement des minima sociaux*/

(npirp15t+npira15t)/(npirp15t+npira15t+npirp16t+npira16t) as  del15_mm, /*delai de traitement<=15*/

nbvim20t/ nbvistot as vis20_mm /*visites 20 min*/

from donnees

order by numorg, annee, mois

;quit;


proc sort data= indicateurs_mm; by numorg annee; run; 

proc transpose data = indicateurs_mm out = transpo_mm (drop=_label_); 

by numorg annee;

id mois;

run;



/*calcul annuel*/


proc sql; 

create table indicateurs_cc as

select numorg, year(ddrefenr) as annee, /*cles de la table*/

sum(ncoua + ncouas) as nbcour_cc, /*nb de courriers*/

sum(npieat) as nbpieces_cc, /*nb pieces*/

sum(nbvistot) as nbvis_cc, /*nb visite*/

sum(telabo+telabos) as nbappabo_cc, /*nb appels aboutis*/

sum(teltragt+teltrsvt)/sum(telabot) as txapptra_cc, /*taux appels traites*/

sum(teltrag+teltrags)/sum(telabag+telabags) as txapptragt_cc, /*taux appels traites agent*/ 

sum(npipso9t)/sum(npipso9t+npipso1t) as minima_cc, /*delai de traitement des minima sociaux*/

sum(npirp15t+npira15t)/sum(npirp15t+npira15t+npirp16t+npira16t) as  del15_cc, /*delai de traitement<=15*/

sum(nbvim20t)/ sum(nbvistot) as vis20_cc /*visites 20 min*/

from donnees

where month(ddrefenr) <=&mv_mois_m0.

group by numorg, year(ddrefenr)

;quit;

                 

/*transposition de la table*/                 

proc sort data= indicateurs_cc; by numorg annee; run; 

proc transpose data = indicateurs_cc out = transpo_cc (drop=_label_); 

by numorg annee;

run;
